     var data = [];
    $(".city").selectOrDie({
        size:4,
        onChange:function () {
            var val = $(".city").val();
            console.log(val);
            var optionhtml = "<option>";
            switch (val){
                case "天津":
                    data=["南开区","红桥区","和平区","河西区"];
                    break;
                case "北京":
                    data=["东城区","西城区","朝阳区"];
                    break;
            }
            $(".district").empty();
            for(var i = 0;i<data.length ;i++){
                optionhtml='<option >'+data[i]+'< / option >';
                $(".district").append( $(optionhtml));
            }
            $(".district").selectOrDie("update");
        }
    });
